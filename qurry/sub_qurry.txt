--SUB QURRY
SELECT
	F.FILM_ID ,
	F.TITLE,
	F.RENTAL_RATE
FROM
	FILM F
WHERE
	F.RENTAL_RATE > (
	SELECT
		AVG(RENTAL_RATE)
	FROM
		FILM F)

--INLINE QURRY
SELECT
	F.FILM_ID ,
	F.TITLE,
	F.RENTAL_RATE
FROM
	FILM F,
	(
	SELECT
		AVG(F.RENTAL_RATE) AS AVG_
	FROM
		FILM F ) B
WHERE
	F.RENTAL_RATE > B.AVG_;
-- SCALAR SUB
SELECT
	F.FILM_ID,
	F.TITLE,
	F.RENTAL_RATE
FROM
	FILM F,
	(
	SELECT
		F.FILM_ID ,
		F.TITLE,
		(
		SELECT
			AVG(F.RENTAL_RATE)
			FROM FILM F ) AS AVG_
	FROM
		FILM F ) B
WHERE
	F.RENTAL_RATE > B.AVG_
